Cria Tabela Cliente
CREATE TABLE CLIENTE (
	ID SERIAL PRIMARY KEY,
	NOME VARCHAR(50),
	EMAIL VARCHAR(50),
	CPF VARCHAR(11),
	ENDERECO VARCHAR(100)
);
Cria Tabela Produto
CREATE TABLE PRODUTO (
	ID SERIAL PRIMARY KEY,
	NOME VARCHAR(50),
	DESCRICAO VARCHAR(50),
	PRECO NUMERIC(10,2),
	QTD_ESTOQUE INTEGER
);
Cria Tabela Relacional Pedido
CREATE TABLE PEDIDO (
	ID SERIAL PRIMARY KEY,
	ID_PRODUTO INTEGER,
	QUANTIDADE INTEGER,
	ID_CLIENTE INTEGER,
	CONSTRAINT FK_PRODUTO FOREIGN KEY (ID_PRODUTO) REFERENCES PRODUTO(ID),
	CONSTRAINT FK_CLIENTE FOREIGN KEY (ID_CLIENTE) REFERENCES CLIENTE(ID)
)
