CREATE TABLE CLIENTE (
	ID SERIAL PRIMARY KEY,
	NOME VARCHAR(50),
	EMAIL VARCHAR(50),
	CPF VARCHAR(11),
	ENDERECO VARCHAR(100)
);
CREATE TABLE PRODUTO (
	ID SERIAL PRIMARY KEY,
	NOME VARCHAR(50),
	DESCRICAO VARCHAR(50),
	PRECO NUMERIC(10,2),
	QTD_ESTOQUE INTEGER
);

CREATE TABLE PEDIDO (
	ID SERIAL PRIMARY KEY,
	ID_CLIENTE INTEGER,
	DATA_CRIACAO TIMESTAMP,
	STATUS VARCHAR(10),
	TOTAL NUMERIC(10,2),
	CONSTRAINT FK_CLIENTE FOREIGN KEY (ID_CLIENTE) REFERENCES CLIENTE(ID)
);

CREATE TABLE ITEM_PEDIDO (
	ID SERIAL PRIMARY KEY,
	ID_PEDIDO INTEGER,
	ID_PRODUTO INTEGER,
	QUANTIDADE INTEGER,
	PRECO NUMERIC(10,2),
	CONSTRAINT FK_PEDIDO FOREIGN KEY (ID_PRODUTO) REFERENCES PEDIDO(ID),
	CONSTRAINT FK_PRODUTO FOREIGN KEY (ID_PRODUTO) REFERENCES PRODUTO(ID)
);
